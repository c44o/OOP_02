# 1.2 Если представлять себе, как эти классы выглядят:
```python
class Rover extends Machine
    def __init__(self):
      self.vehicle_type = "rover"
      self.mass_kg = 8.5
      self.num_wheels = 6
      self.mode = "MANUAL"
      self.armed = False
      self.accelerometer = Accelerometer()
      self.battery = LiIon18650()

      def arm(self):
          if not self.battery.is_low and self.accelerometer.healthy:
              self.armed = True
  
      def start(self):
          if self.armed:
              self.accelerometer.start()

class Accelerometer:
    def __init__(self):
        self.needs_calibration = True
        self.sample_rate_hz = 200
        self.healthy = True

class LiIon18650:
    def __init__(self):
        self.model = "Li-Ion"
        self.capacity_mah = 3500
        self.current_voltage = 4.1
        self.is_low = False
```

# 1.3 Пример побочного эффекта:
```python
# объект батарейки
class LiIon18650:
    def __init__(self, capacity_mah=3500, current_voltage_v=4.1, is_low=False):
        self.chemistry = "Li-Ion"
        self.capacity_mah = capacity_mah
        self.nominal_voltage_v = 3.6
        self.is_low = is_low

    def __str__(self):
        return (f"LiIon18650(voltage={self.current_voltage_v:.2f}V, "
                f"is_low={self.is_low}, capacity={self.capacity_mah}mAh)")
# объект ровера (str чтобы что-то было видно к консоли)
class Rover:
    def __init__(self, name, battery):
        self.name = name
        self.battery = battery

    def __str__(self):
        return f"Rover(name={self.name}, battery={self.battery})"

# функция чтобы снять часть заряда с батарейки
def change_voltage(battery: LiIon18650, drop: float):
    battery.current_voltage_v -= drop
    if battery.current_voltage_v < 3.5:
        battery.is_low = True

# создаём батарейку, одну, и две разных машинки
main_battery = LiIon18650(current_voltage_v=3.8, is_low=False)

rover_a = Rover("A", main_battery)
rover_b = Rover("B", main_battery)

print(rover_a)
print(rover_b)

# батарейка садится у обеих машинок сразу
simulate_voltage_drop(rover_a.battery, drop=0.6)
```
