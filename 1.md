# 1.1 Пример - Ardupilot, набор модулей для управления БПЛА, написан на C++.
Предположительные основные классы 
(для описания железа)
- Machine (Vehicle/Drone/Copter..)
- Sensor (и подклассы - Gyroscope, Barometer, GPS, Lidar, Sonar, Radar)
- Motor (Actuator/Rotor)
- Battery
- GroundControlStation
# 1.2 Если представлять себе, как эти классы выглядят:
```python
class Rover extends Machine
    def __init__(self):
      self.vehicle_type = "rover"
      self.mass_kg = 8.5
      self.num_wheels = 6
      self.mode = "MANUAL"
      self.armed = False
      self.accelerometer = Accelerometer()
      self.battery = LiIon18650()

      def arm(self):
          if not self.battery.is_low and self.accelerometer.healthy:
              self.armed = True
  
      def start(self):
          if self.armed:
              self.accelerometer.start()

class Accelerometer:
    def __init__(self):
        self.needs_calibration = True
        self.sample_rate_hz = 200
        self.healthy = True

class LiIon18650:
    def __init__(self):
        self.model = "Li-Ion"
        self.capacity_mah = 3500
        self.current_voltage = 4.1
        self.is_low = False
```
