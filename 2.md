1. 2 класса с методами
```python
class Machine:
    def __init__(self):
        self.mass_kg = 10

class Rover(Machine):
    def __init__(self):
      super().__init__()
      self.vehicle_type = "rover"
      self.mass_kg = 8.5
      self.num_wheels = 6
      self.mode = "MANUAL"
      self.armed = False
      self.accelerometer = Accelerometer()
      self.battery = LiIon18650()

    def arm(self):
        if not self.battery.is_low and self.accelerometer.healthy:
            self.armed = True

    def disarm(self):
        self.armed = False

    def start(self):
        if self.armed:
            self.accelerometer.start()

    def stop(self):
        self.accelerometer.stop()

class Accelerometer:
    def __init__(self):
        self.needs_calibration = True
        self.sample_rate_hz = 200
        self.healthy = True
        self.is_working = False

    def start(self):
        self.is_working = True

    def stop(self):
        self.is_working = False

class LiIon18650:
    def __init__(self):
        self.model = "Li-Ion"
        self.capacity_mah = 3500
        self.current_voltage = 4.1
        self.is_low = False
```
